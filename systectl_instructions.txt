To create this file:

sudo nano /etc/systemd/system/muiva.service
-----------

[Unit]
Description=Streamlit Muiva
After=network.target

[Service]
ExecStart=/home/ubuntu/miniconda/bin/streamlit run /home/ubuntu/muiva_app/Inicio.py
WorkingDirectory=/home/ubuntu/muiva_app
User=ubuntu
Group=ubuntu
Restart=always

[Install]
WantedBy=multi-user.target

---------------
sudo systemctl daemon-reload
sudo systemctl enable muiva.service
sudo systemctl start muiva.service

-----------------
To enable:
sudo systemctl enable muiva.service

To start:
sudo systemctl start muiva.service

To stop:
sudo systemctl stop muiva.service

To status:
sudo systemctl status muiva.service

To reload:
sudo systemctl daemon-reload

To delete:
sudo systemctl stop muiva.service
sudo systemctl disable muiva.service
sudo rm /etc/systemd/system/muiva.service
sudo systemctl daemon-reload
